void UBFL_ManualAsset::EditBlueprintProperty(UBlueprint* Blueprint)
{
	//UObject* CDO = Blueprint->GeneratedClass.GetDefaultObject();
	//if (AActor* Actor = Cast<AActor>(CDO)) {
	//	Actor->SetHidden(true);
	//}


	UBlueprintGeneratedClass* GeneratedClass = Cast<UBlueprintGeneratedClass>(Blueprint->GeneratedClass);
	if (GeneratedClass) {
		TArray<USCS_Node*> AllNodes = GeneratedClass->SimpleConstructionScript->GetAllNodes();
		for (USCS_Node* Node : AllNodes) {
			UActorComponent* ComponentTemplate = Node->GetActualComponentTemplate(GeneratedClass);
			if (UStaticMeshComponent* smc = Cast<UStaticMeshComponent>(ComponentTemplate)) {
				// sm->ComponentTags.Add(FName("Edited"));
				smc->ComponentTags.Empty();
				UStaticMesh* sm = smc->GetStaticMesh();

				if (sm) {
					UPackage* Package = sm->GetOutermost();
					if (Package) {
						UMetaData* MetaData = Package->GetMetaData();
						if (MetaData) {
							FString Value = MetaData->GetValue(sm, TEXT("FBX.PhysicsMaterialType"));
							smc->ComponentTags.Add(FName(Value));
						}
					}
				}
			}
		}
	}

	// 重新编译蓝图
	// FKismetEditorUtilities::CompileBlueprint(Blueprint);
}
