using System.Collections;
using System.Collections.Generic;
using UnityEngine;


[ExecuteInEditMode]
[RequireComponent(typeof(Camera))]
public class PostFX_Glitch : MonoBehaviour
{
    // Scan line jitter

    [ Range(0, 1)] public float _ScanLineJitter = 0;

    // Vertical jump

    [ Range(0, 1)] public float VerticalJump = 0;
  
    // Horizontal shake

    [ Range(0, 1)] public float HorizontalShake = 0;

    // Color drift

    [Range(0, 1)] public float _colorDrift = 0;
    
    Material mat;
    
    float VerticalJumpTime;

    private static int ID_ScanLineJitter = Shader.PropertyToID("_ScanLineJitter");
    private static int ID_VerticalJump = Shader.PropertyToID("_VerticalJump");
    private static int ID_HorizontalShake = Shader.PropertyToID("_HorizontalShake");
    private static int ID_ColorDrift = Shader.PropertyToID("_ColorDrift");
        
    void OnRenderImage(RenderTexture source, RenderTexture destination)
    {
        if (mat == null)
        {
            // Hidden/ 下的 shader 只能用 shader.find 找到
            mat = new Material(Shader.Find("Hidden/TAPro_Glitch01"));
        }
        // 把一些计算放到 cpu 减少 gpu 负担，也更节省
        var scanline_threshold = Mathf.Clamp01(1.0f - _ScanLineJitter * 1.2f);
        var scanline_amount = 0.002f + Mathf.Pow(_ScanLineJitter, 3) * 0.05f;
        mat.SetVector(ID_ScanLineJitter, new Vector2(scanline_amount, scanline_threshold));
        
        VerticalJumpTime += Time.deltaTime * VerticalJump * 11.3f;
        var vj = new Vector2(VerticalJump, VerticalJumpTime);
        mat.SetVector(ID_VerticalJump, vj);

        mat.SetFloat(ID_HorizontalShake, HorizontalShake * 0.2f);

        var cd = new Vector2(_colorDrift * 0.04f, Time.time * 606.11f);
        mat.SetVector(ID_ColorDrift, cd);

        Graphics.Blit(source, destination, mat);
    }
}

Shader "Hidden/TAPro_Glitch01"
{
    Properties
    {
        _MainTex ("-", 2D) = "" {}
    }
    CGINCLUDE
    #include "UnityCG.cginc"

    sampler2D _MainTex;
    float2 _MainTex_TexelSize;

    float2 _ScanLineJitter; // (displacement, threshold)
    float2 _VerticalJump; // (amount, time)
    float _HorizontalShake;
    float2 _ColorDrift; // (amount, time)

    float nrand(float x, float y)
    {
        return frac(sin(dot(float2(x, y), float2(12.9898, 78.233))) * 43758.5453);
    }

    half4 frag(v2f_img i) : SV_Target
    {
        float u = i.uv.x;
        float v = i.uv.y;

        //扫描线 scan line
        float jitter = nrand(v, _Time.x) * 2 - 1;
        jitter *= step(_ScanLineJitter.y, abs(jitter)) * _ScanLineJitter.x;
        
        //垂直跳动
        float jump = lerp(v, frac(v + _VerticalJump.y), _VerticalJump.x);

        //水平跳动
        float shake = (nrand(_Time.x, 2) - 0.5) * _HorizontalShake;

        //颜色分离
        float drift = sin(jump + _ColorDrift.y) * _ColorDrift.x;

        half4 src1 = tex2D(_MainTex, frac(float2(u + jitter + shake, jump)));
        half4 src2 = tex2D(_MainTex, frac(float2(u + jitter + shake + drift, jump)));

        return half4(src1.r, src2.g, src1.b, 1);
    }
    ENDCG
    SubShader
    {
        Pass
        {
            ZTest Always Cull Off ZWrite Off
            CGPROGRAM
            // 默认使用 unity 自己的咯
            #pragma vertex vert_img
            #pragma fragment frag
            ENDCG
        }
    }
}
