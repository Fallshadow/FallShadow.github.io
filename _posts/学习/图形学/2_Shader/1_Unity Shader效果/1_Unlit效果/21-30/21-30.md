```Cpp
float3 L_Point_Dir = _Light021PointPos - pixel.worldPos;
float3 L_Point = normalize(L_Point_Dir);
float distance = length(L_Point_Dir);
float linearFade = 1 - saturate(distance _Light021PointColorAndRange.w);
linearFade = pow(linearFade,_LightPointLinearFadeSoft);
float4 wrapLambert_P = Diffuse(N, L_Point,_WrapLight);
float4 speculary_P = Speculary(N, L_Point, V,_VHMult,_BlinPhong_Exp, _BlinPhong_Scale);
float4 blinnPhong_P = BlinnPhong(N, L_Point, V,_WrapLight, _VHMult,_BlinPhong_Exp,_BlinPhong_Scale);
if (_LightType == 0)
{
    // return frac(_Light021PointColorAndRange.xyzz * _Light021PointColorAndRange.w * _Light021PointPos.xyzz);
    return _Light021PointColorAndRange.xyzz * linearFade * baseMap * wrapLambert_P * 10;
}
else if (_LightType == 1)
{
    // 距离平方衰减！
    float atten = 1.0 / max(distance * distance, 0.01);
    
    return _Light021PointColorAndRange.xyzz * atten * linearFade * baseMap * blinnPhong_P;
}
```
